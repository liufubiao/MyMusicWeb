    private static void dateTimeFormatter() {
        /**======pattern格式 模式*/
        ZonedDateTime zonedDateTime = ZonedDateTime.now();
        LocalDate localDate = LocalDate.now();
        LocalTime localTime = LocalTime.now();
        LocalDateTime localDateTime = LocalDateTime.now();
        /*-------------------------------------------------*/
        /**格式化时间对象localdate localtime localdatetime zonedatetime*/
        DateTimeFormatter forMat = DateTimeFormatter.ofPattern("yyyy年MM月dd日 HH时mm分ss秒");
        DateTimeFormatter yMdFormat = DateTimeFormatter.ofPattern("yyyy年MM月dd日");
        DateTimeFormatter forMat1 = DateTimeFormatter.ofPattern("HH时mm分ss秒");
        String format1 = zonedDateTime.format(forMat);
        System.out.println(format1);
        String format2 = localDate.format(yMdFormat);//无法使用"yyyy年MM月dd日 HH时mm分ss秒"的格式化器进行格式化 因为localDate没有时分秒毫秒
        System.out.println(format2);
        String format3 = localTime.format(forMat1);//无法使用"yyyy年MM月dd日 HH时mm分ss秒"的格式化器进行格式化 因为localtime没有年月日
        System.out.println(format3);
        /**解析时间字符串   时间字符串解析为时间对象*/
        LocalDateTime parse = LocalDateTime.parse(format1, forMat);
        System.out.println("parse解析后的时间对象"+parse);

    private static void localDateTime() {
        /**----------------java 日期时间  LocalDateTime--------------*/
        LocalDateTime now = LocalDateTime.now();//当前时间日期 年月日 时分秒毫秒
        LocalDateTime of = LocalDateTime.of(2001, 6, 13, 1, 30, 59, 5000000);
        System.out.println(of);
        /**------------------get---------------------*/
        now.getYear();//年
        now.getMonthValue();//月
        now.getDayOfMonth();//日
        now.getDayOfYear();//日
        now.getDayOfWeek();//日
        now.getHour();//时
        now.getMinute();//分
        now.getSecond();//秒
        now.getNano();//纳秒 （以毫秒显示）
        /**------plus / minus--------------------*/
        LocalDateTime localDateTime = now.plusYears(1);
        LocalDateTime localDateTime1 = now.plusNanos(5000000);
        LocalDateTime localDateTime2 = now.minusYears(2);
        //增加或者减少时间日期 减少年月日 时分秒纳秒
        /**-------------------equals isafter isbefore ---------------------*/
        LocalDateTime plusWeeks = now.plusWeeks(1);
        LocalDateTime minusWeeks = now.minusWeeks(2);
        System.out.println(plusWeeks.isAfter(minusWeeks));
        System.out.println(plusWeeks.isBefore(minusWeeks));
        System.out.println(plusWeeks.equals(minusWeeks));


    }

    private static void localDate(LocalDate dateyMd, LocalDate date) {
        /**==============获取日期的年月日的值======LocalDate======*/
        int year = dateyMd.getYear();//获取年份
        System.out.println("年:"+year);
        int monthValue = dateyMd.getMonthValue();//获取月份值
        System.out.println("Month:"+monthValue);

        DayOfWeek dayOfWeek = dateyMd.getDayOfWeek();
        System.out.println("一个月的第几天："+dayOfMonth+"一年的第几天："+dayOfYear+"一个星期的第几天："+dayOfWeek.getValue());
        /**----with系列的修改日期的方法---------minute分钟 minus减去*/
        LocalDate date1 = dateyMd.withYear(2023);
        System.out.println("with修改年份后的日期："+date1);
        LocalDate date2 = dateyMd.withMonth(7);
        LocalDate date3 = dateyMd.withDayOfMonth(5);
        System.out.println(date2+"\r\n"+date3);
        /**------------增加/减少日期---------------*/
        LocalDate date4 = date2.plusMonths(2);//增加了2个月 2月-》4月 2+2
        LocalDate date5 = date2.plusYears(1);//增加年份
        LocalDate date6 = date2.plusDays(10);//增加天数
        LocalDate date7 = date2.minusMonths(2);//减少了2个月 2月-》4月 2+2
        LocalDate date8 = date2.minusYears(1);//减少年份
        LocalDate date9 = date2.minusDays(10);//增加天数
        /**------判断日期的先后/和是否相等  after before equals------*/
        LocalDate plusdate = dateyMd.plusDays(10);//+10day
        LocalDate minusdate = dateyMd.minusDays(10);//-10day
        /**日期是否相等 同一天*/ //equals
        System.out.println(plusdate.equals(minusdate));//不是同一天
        System.out.println(plusdate.isAfter(minusdate));//plusdate在minusdate之后 true
        System.out.println(minusdate.isBefore(plusdate));//minusdate在plusdate之前 true
    }
