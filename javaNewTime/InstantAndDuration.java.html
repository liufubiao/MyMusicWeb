import java.time.*;

public class InstantAndDuration {
    /**时间戳 和时间间隔 持续的时间*/
    public static void main(String[] args) {
        instant();//计算总秒数的时间线/戳
        period();//计算时间段 时间间隔
        duration();//计算持续时间  天时分秒毫秒纳秒的间隔计算
        /**
         * public static Duration between(开始时间对象1,截止时间对象2)
         */
    }

    private static void duration() {
        /**计算一个时间对象持续到另一个时间对象的持续时间*/
        /**计算两个时间对象相差的天数、小时数、分数、秒数、纳秒数；支持LocalTime、LocalDateTime、Instant等
         *
         */
        LocalDateTime localDateTime = LocalDateTime.of(2001, 6, 13, 1, 32, 55, 4000000);
        LocalDateTime now = LocalDateTime.now();
        Duration between = Duration.between(localDateTime, now);
        long days = between.toDays();//持续的天数
        System.out.println(days);
        long hours = between.toHours();
        System.out.println(hours);//持续的小时数
        long minutes = between.toMinutes();//持续的分钟
        System.out.println(minutes);


    }

    private static void period() {
        /**-------------------可以用于计算两个 LocalDate对象 相差的年数、月数、天数--*/
        LocalDateTime localDateTime = LocalDateTime.of(2001, 6, 13, 1, 32, 55, 4000000);
        LocalDateTime now = LocalDateTime.now();
        LocalDate localDate = localDateTime.toLocalDate();
        LocalDate localDate1 = now.toLocalDate();
        Period between = Period.between(localDate, localDate1);
        int years = between.getYears();
        System.out.println(years);
        int months = between.getMonths();
        System.out.println(months);
        int days = between.getDays();
        System.out.println(days);
    }

    private static void instant() {
        /**------Instant对象的作用：做毫秒级代码的性能分析，或者记录用户的操作时间点----------*/
        Instant instant = Instant.now();//获取时间戳/线 对象
        System.out.println(instant.getEpochSecond());/**获取从1970年 1月1日 00:00:00 到现在的总秒数*/
        long epochSecond1 = instant.getEpochSecond();
        System.out.println(instant.getNano());//不到一秒的纳秒数 909000000纳秒 909毫秒
        int nano1 = instant.getNano();
        /**
         * 运行的代码============运行的代码
         * 运行的代码============运行的代码
         * 运行的代码============运行的代码
         * 运行的代码============运行的代码
         * 运行的代码============运行的代码
         */
        Instant instant1 = Instant.now();//获取时间戳/线 对象
        long epochSecond = instant1.getEpochSecond();
        int nano = instant1.getNano();
        System.out.println("程序运行的时间："+(epochSecond-epochSecond1)+"."+(nano-nano1));

    }
}
